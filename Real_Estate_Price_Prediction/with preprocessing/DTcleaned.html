from sklearn.metrics import r2_score, mean_absolute_error, mean_squared_error  # Import additional metrics
from sklearn.metrics import make_scorer  # Import make_scorer for custom scoring functions

# Define a custom scoring function for MAE
def custom_mae(y_true, y_pred):
    return mean_absolute_error(y_true, y_pred)

# Define a custom scoring function for MSE
def custom_mse(y_true, y_pred):
    return mean_squared_error(y_true, y_pred)

# Use the custom scoring functions with make_scorer
mae_scorer = make_scorer(custom_mae, greater_is_better=False)
mse_scorer = make_scorer(custom_mse, greater_is_better=False)

# Perform 10-fold cross-validation on the model
